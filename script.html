const API = "https://fetvaci.YOUR-WORKER.workers.dev"; // 🔎 SEARCH document.getElementById("search")?.addEventListener("input", async (e) => { const q = e.target.value; const res = await fetch(`${API}/api/search?q=${q}`); const data = await res.json(); document.getElementById("recommended").innerHTML = data.map(x => card(x)).join(""); }); // 🏠 HOME LOAD async function loadHome() { const res = await fetch(`${API}/api/home`); const data = await res.json(); document.getElementById("recommended").innerHTML = data.recommended.map(card).join(""); document.getElementById("controversial").innerHTML = data.controversial.map(cardMekruh).join(""); } // 📦 CARD function card(x) { return `

${x.soru}

`; } // ⚠️ MEKRUH CARD function cardMekruh(x) { return `

${x.soru}

${x.hukum}

`; } // 📄 POST LOAD async function loadPost() { const slug = new URLSearchParams(location.search).get("slug"); const res = await fetch(`${API}/api/post/${slug}`); const data = await res.json(); document.getElementById("post").innerHTML = `

${data.soru}

Hüküm

${data.hukum}

${data.ayet_metin ? `

Kur'an

${data.ayet_metin}
${data.ayet_referans}
` : ""} ${data.hadis_metin ? `

Hadis

${data.hadis_metin}
${data.hadis_bilgi}
` : ""}

Mezhepler

${renderMezhep(data.mezhepler_json)}

Ansiklopedik Açıklama

${data.aciklama}

`; } // 🎨 MEZHEP TABLO function renderMezhep(json) { const m = typeof json === "string" ? JSON.parse(json) : json; return `
Hanefi: ${m.Hanefi}
Şafii: ${m.Şafii}
Maliki: ${m.Maliki}
Hanbeli: ${m.Hanbeli}
`; } loadHome(); loadPost();
